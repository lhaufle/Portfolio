CREATE TABLE Users (
   ID INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
   UserName VARCHAR(50) NOT NULL,
   Pass VARCHAR(15) NOT NULL,
   Privilege VARCHAR(30) NOT NULL,
   TimeAccess TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
   )CHARACTER SET utf8;
   
CREATE TABLE SessionComments (
	ID INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	name VARCHAR(45),
	rating VARCHAR(20) NOT NULL,
	comment VARCHAR(300) NOT NULL, 
	user_id INT UNSIGNED NOT NULL,
	TimeAccess TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
	FOREIGN KEY fk_comment(user_id)
    REFERENCES Users(ID)
) CHARACTER SET utf8;

CREATE TABLE ProductComments (
	ID INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	name VARCHAR(45),
	rating VARCHAR(20) NOT NULL,
	comments VARCHAR(300) NOT NULL, 
	user_id INT UNSIGNED NOT NULL,
	TimeAccess TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
	FOREIGN KEY fk_comment(user_id)
    REFERENCES Users(ID)
) CHARACTER SET utf8;

CREATE TABLE Items (
	ID INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	img VARCHAR(60) NOT NULL,
	name VARCHAR(30) NOT NULL,
    type VARCHAR(30) NOT NULL,	
	price FLOAT(8, 2)
) CHARACTER SET utf8;

CREATE TABLE Cart (
	ID INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	name VARCHAR(30) NOT NULL,
	price FLOAT(8,2),
	due FLOAT(8,2),
	qty INT UNSIGNED NOT NULL
) CHARACTER SET utf8;

CREATE TABLE HomePage(
	ID INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	header varchar(50) NOT NULL, 
	body varchar(500) NOT NULL,
	changeID INT UNSIGNED NOT NULL,
	timeChanged TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
	FOREIGN KEY fk_home(changeID) REFERENCES Users(ID)
) CHARACTER SET utf8

CREATE TABLE Charity(
	ID INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	header varchar(50) NOT NULL, 
	body varchar(500) NOT NULL,
	changeID INT UNSIGNED NOT NULL,
	timeChanged TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
	FOREIGN KEY fk_charity(changeID) REFERENCES Users(ID)
) CHARACTER SET utf8

CREATE TABLE Facts (
	ID INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	fact VARCHAR(50) NOT NULL, 
	description VARCHAR(200)
)CHARACTER SET utf8

CREATE TABLE LoginTracker (
	ID INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	userID INT UNSIGNED NOT NULL,
    timeLogged TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
	FOREIGN KEY fk_logged(userID) REFERENCES Users(ID)
)CHARACTER SET utf8
